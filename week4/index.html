let students = ["Bobby Teenager", "Ellen Doe", "Jerome Boggs", "Philipe Philips", "Cher Smith"];

function pickAStudent(students){
  let whichNum = Math.floor(Math.random()*students.length);
  let student = students[whichNum];
  
  return student;
}

let whichStudent = pickAStudent(students);
console.log("Selected student: ",whichStudent);

Anonymous Functions

  1. Anonymous functions are created at runtime
  2. They don’t have a name

Example of a named (not anonymous function):

function flyToTheMoon(){
      alert("Zoom! Zoom! Zoom!");
}

flyToTheMoon();

Anonymously

function() {
   alert("Zoom! Zoom! Zoom!");
}

So we have an anonymous function stored as flyToTheMoon now. How do we run it if it doesn’t have a name?

Debugging

We can use the Inspect tool/browser tools to view the DOM, applied CSS, and click through to the JavaScript console, and the responsive browser mode.

The rendered DOM code will appear to show you how your JavaScript and CSS was applied to your HTML.

The applied CSS tool will allow you to select HTML elements and see what CSS was applied. You can click the checkbox to turn on and off certain CSS attributes.

The JavaScript console is useful for seeing error messages, warnings and debugging statements such as console.log().

The responsive browser mode serves as an emulator to let you preview a website as if you are using a device such as a phone or tablet to see how it would appear.
